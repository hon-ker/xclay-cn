python
import requests
from bs4 import BeautifulSoup
from urllib.parse import urlparse, urljoin


def crawl_gif_images(url, max_depth, current_depth=1):
    # 用于存储找到的 GIF 图片 URL 的列表
    gif_urls = []

    # 获取页面内容
    response = requests.get(url)
    if response.status_code == 200:
        soup = BeautifulSoup(response.content, 'html.parser')

        # 查找页面中的 GIF 图片标签（通常是 <img>）
        gif_tags = soup.find_all('img', attrs={'src': lambda x: x.endswith('.gif')})

        # 提取 GIF 图片的 URL
        for tag in gif_tags:
            gif_url = tag['src']
            gif_urls.append(gif_url)

        # 递归爬取更深层级的 URL
        if current_depth < max_depth:
            # 查找页面中的其他链接并递归爬取
            link_tags = soup.find_all('a')
            for link_tag in link_tags:
                href = link_tag.get('href')
                if href:
                    next_url = urljoin(url, href)
                    next_domain = urlparse(next_url).netloc
                    current_domain = urlparse(url).netloc
                    # 确保只爬取同一域名下的链接
                    if next_domain == current_domain:
                        gif_urls += crawl_gif_images(next_url, max_depth, current_depth + 1)

    return gif_urls
